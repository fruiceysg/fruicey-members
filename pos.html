function buildReceiptLines(){

  var sale = _lastSale;

  if(!sale) return '';

  var isO = ONLINE_PLATFORMS.includes(sale.payment);

  var W = 32;

  function pad(s,n){ s=String(s); while(s.length<n) s+=' '; return s.slice(0,n); }

  function center(s){ var p=Math.max(0,Math.floor((W-s.length)/2)); return ' '.repeat(p)+s; }

  function right(l,r){ var sp=Math.max(1,W-l.length-r.length); return l+' '.repeat(sp)+r; }

  function dashes(){ return '-'.repeat(W); }


  var lines = [

    center('FRUICEY'),

    center('68 Jalan Jurong Kechil'),

    center('#01-10 Singapore 596180'),

    center('UEN: 53434968D'),

    center('Tel: 8509 8095'),

    dashes(),

    right('Receipt:', sale.rcptNo||''),

    right('Date:', sale.date),

    right('Time:', sale.time)

  ];

  if(sale.member) lines.push(right('Member:', sale.member));

  lines.push(dashes());

  sale.items.forEach(function(item, i){

    var amt = isO ? '$0.00' : '$'+(item.price*item.qty).toFixed(2);

    lines.push((i+1)+'. '+(item.label||'').slice(0,W-3));

    lines.push(right('  x'+item.qty, amt));

  });

  lines.push(dashes());

  if(sale.discount>0){

    lines.push(right('Subtotal','$'+sale.subtotal.toFixed(2)));

    lines.push(right('Discount','-$'+sale.discount.toFixed(2)));

  }

  lines.push(right('TOTAL','$'+sale.total.toFixed(2)));

  lines.push(right('Payment', sale.payment));

  if(sale.payment==='Cash'&&sale.cashTendered!=null){

    lines.push(right('Cash','$'+(sale.cashTendered||0).toFixed(2)));

    lines.push(right('Change','$'+(sale.changeDue||0).toFixed(2)));

  }

  lines.push(dashes());

  lines.push(center('Thank You For Visiting'));

  lines.push(center('Fruicey! www.fruicey.sg'));

  lines.push(center('Join Rewards - Free!'));

  lines.push(center('fruiceysg.github.io/'));

  lines.push(center('fruicey-members/stamp.html'));

  lines.push('');

  lines.push('');

  return lines.join('\n');

}


function printReceipt(){

  var sale = _lastSale;

  if(!sale){ showToast('No receipt to print'); return; }

  var txt = buildReceiptLines();


  // Try Sunmi native APIs

  if(window.SunmiPrinter){

    try{ window.SunmiPrinter.printerInit(); window.SunmiPrinter.printText(txt); window.SunmiPrinter.paperFeed(3); showToast('Printing...'); return; }catch(e){}

  }

  if(window.sunmiInnerPrinter){

    try{ sunmiInnerPrinter.printerInit(null); sunmiInnerPrinter.printText(txt,null); sunmiInnerPrinter.autoOutPaper(null); showToast('Printing...'); return; }catch(e){}

  }

  // Try Sunmi device IP + common Xilnex/ESC-POS ports

  var SUNMI_IP = '10.150.255.157';

  var endpoints = [

    'http://'+SUNMI_IP+':10443/print',

    'http://'+SUNMI_IP+':8080/print',

    'http://'+SUNMI_IP+':9100',

    'http://'+SUNMI_IP+':3000/print',

    'http://'+SUNMI_IP+':8008/print',

    'http://localhost:10443/print',

    'http://localhost:9100/print'

  ];

  var tried = 0;

  function tryEndpoint(){

    if(tried >= endpoints.length){ window.print(); return; }

    var url = endpoints[tried];

    fetch(url, {

      method:'POST',

      mode:'no-cors',

      headers:{'Content-Type':'text/plain'},

      body: txt

    }).then(function(){

      showToast('Sent to Sunmi printer!');

    }).catch(function(){

      tried++;

      tryEndpoint();

    });

  tryEndpoint();

async function pollNewAppOrders(){

    const res = await apiGet({action:'getOrders'});

    if(res.error || !res.orders) return;

    const orders = res.orders;

    if(!_seenOrderIdsInitialized){

      // First run — record existing orders without alerting, so old orders don't trigger a false alarm

      orders.forEach(function(o){ _seenOrderIds.add(o.id); });

      _seenOrderIdsInitialized = true;

      return;

    }

    const freshOrders = orders.filter(function(o){ return !_seenOrderIds.has(o.id); });

    if(freshOrders.length){

      freshOrders.forEach(function(o){ _seenOrderIds.add(o.id); });

function computeTier(member,txns){

  const TIERS=[{name:'Member',min:0,discount:0},{name:'Gold',min:50,discount:5},{name:'Platinum',min:100,discount:10},{name:'Diamond',min:400,discount:15}];

  const now=new Date();

  let ps=member.tierSince?new Date(member.tierSince):new Date(member.since);

  let pe=new Date(ps);pe.setMonth(pe.getMonth()+2);

  while(pe<=now){ps=new Date(pe);pe=new Date(ps);pe.setMonth(pe.getMonth()+2);}

  const spend=(txns||[]).filter(t=>t.memberId===member.id&&new Date(t.date)>=ps&&new Date(t.date)<pe).reduce((s,t)=>s+parseFloat(t.amount||0),0);

  return TIERS.slice().reverse().find(t=>spend>=t.min)||TIERS[0];

}

function cleanSaleDate(s){

  // Fix date if it comes as ISO string from Sheets (e.g. "2026-06-29T00:00:00.000Z")

  if(s.date && s.date.includes('T')){

    // Convert UTC ISO to SGT date string

    const d = new Date(s.date);

    const sgt = new Date(d.getTime() + 8*3600*1000);

    s.date = sgt.toISOString().slice(0,10);

  }

  // Fix time if it comes as decimal fraction from Sheets (e.g. 0.625 = 15:00)

  if(s.time && typeof s.time === 'number' && s.time < 1){

    const totalMins = Math.round(s.time * 24 * 60);

    const h = Math.floor(totalMins / 60);

    const m = totalMins % 60;

    s.time = String(h).padStart(2,'0') + ':' + String(m).padStart(2,'0');

  }

  // Fix time if it comes as ISO string — extract HH:MM directly, no offset

  if(s.time && typeof s.time === 'string' && s.time.includes('T')){

    s.time = s.time.slice(11,16);

  }

  return s;

}


async function loadSalesFromSheets(){

  // Show loading state

  const sumEl = document.getElementById('sales-summary');

  if(sumEl) sumEl.innerHTML = '<div style="text-align:center;padding:20px;color:#888">Syncing from server...</div>';


  try{

    const res = await apiGet({action:'getPOSSales'});

    console.log('getPOSSales response:', res);


    if(res.error){

      showToast('Sync error: '+res.error);

      console.error('getPOSSales error:', res.error);

    } else if(res.sales && res.sales.length){

      const sheetSales = res.sales.map(s=>{

        try{ s.items = typeof s.items==='string' ? JSON.parse(s.items) : (s.items||[]); }catch(e){ s.items=[]; }

        s.total    = parseFloat(s.total||0);

        s.subtotal = parseFloat(s.subtotal||0);

        s.discount = parseFloat(s.discount||0);

        s.voided   = s.voided==='true';

        return cleanSaleDate(s);

      });

      // Sheets is the single source of truth - replace local entirely

      salesLog = sheetSales;

      try{ localStorage.removeItem('fruicey_pos_sales'); }catch(e){} // no local cache

      showToast('Synced '+sheetSales.length+' transactions'); console.log('Sales from Sheets:', sheetSales);

    } else if(res.sales && res.sales.length === 0) {

      // Sheets is empty - clear local cache too

      salesLog = [];

      try{ localStorage.removeItem('fruicey_pos_sales'); }catch(e){}

      showToast('Sales history up to date');

    } else {

      showToast('Unexpected response from Sheets');

      console.log('Full response:', res);

    }

  }catch(e){

    console.error('loadSalesFromSheets failed:', e);

    showToast('Could not connect to server');

  }

  renderSalesReport();

}

async function loadStock(){

  try{

    const res = await apiGet({action:'getStock'});

    if(res.stock){

      // Initialize fresh stock status object

      stockStatus = {};

      

      // Ensure all values are proper booleans

      Object.keys(res.stock).forEach(id => {

        const val = res.stock[id];

        // Parse any representation (true, 'true', false, 'false') to boolean

        stockStatus[id] = (val === true || val === 'true');

      });

    }

  }catch(e){ console.log('Could not load stock'); }

  renderProducts();

}


async function toggleStock(pid, available){

  stockStatus[pid] = available;

  renderProducts(); // instant UI update

  try{

    await apiPost({action:'updateStock', id:pid, available});

    showToast((available ? 'Back in stock' : 'Marked out of stock') + ': ' + (PRODUCT_MAP[pid]||{name:pid}).name);

  }catch(e){ showToast('Could not update stock'); }

}

  if(!card || e.target.closest('.stock-toggle-btn')) return;

  if(isAvail){

    }

  }

});
